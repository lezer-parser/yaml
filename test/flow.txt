# Simple flow values

{foo: [bar, baz],
 bar: "one
two three"}

==>

Stream(Document(FlowMapping(
  Pair(Literal,FlowSequence(Item(Literal),Item(Literal))),
  Pair(Literal,QuotedLiteral))))

# Pairs

[one: two, ? : four, x: ]

==>

Stream(Document(FlowSequence(
  Item(FlowMapping(Pair(Literal,Literal))),
  Item(FlowMapping(Pair(Literal))),
  Item(FlowMapping(Pair(Literal))))))
